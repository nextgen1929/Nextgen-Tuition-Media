        // 📚 টিউটর লোড
        function loadTutors() {
            db.collection('tutors').orderBy('createdAt', 'desc').onSnapshot((snapshot) => {
                const tbody = document.getElementById('tutorsTableBody');
                tbody.innerHTML = '';
                let i = 1;
                snapshot.forEach((doc) => {
                    const data = doc.data();
                    const statusClass = data.status === 'Active' ? 'badge-active' : 'badge-inactive';
                    tbody.innerHTML += `
                        <tr>
                            <td>${i++}</td>
                            <td>${data.name}</td>
                            <td>${data.subject}</td>
                            <td><span class="badge-status ${statusClass}">${data.status}</span></td>
                            <td>
                                <button class="action-btn-sm btn-edit" onclick="editTutor('${doc.id}')"><i class="fa-regular fa-pen-to-square"></i></button>
                                <button class="action-btn-sm btn-delete" onclick="deleteTutor('${doc.id}')"><i class="fa-regular fa-trash-can"></i></button>
                            </td>
                        </tr>
                    `;
                });
                updateStats();
            });
        }

        // 🎓 স্টুডেন্ট লোড
        function loadStudents() {
            db.collection('students').orderBy('createdAt', 'desc').onSnapshot((snapshot) => {

                tbody.innerHTML = '';
                let i = 1;
                snapshot.forEach((doc) => {
                    const data = doc.data();
                    const statusClass = data.status === 'Active' ? 'badge-active' : data.status === 'Pending' ? 'badge-pending' : 'badge-inactive';
                    tbody.innerHTML += `
                        <tr>
                            <td>${i++}</td>
                            <td>${data.name}</td>
                            <td>${data.class}</td>
                            <td>${data.tutor}</td>
                            <td><span class="badge-status ${statusClass}">${data.status}</span></td>
                            <td>
                                <button class="action-btn-sm btn-edit" onclick="editStudent('${doc.id}')"><i class="fa-regular fa-pen-to-square"></i></button>
                                <button class="action-btn-sm btn-delete" onclick="deleteStudent('${doc.id}')"><i class="fa-regular fa-trash-can"></i></button>
                            </td>
                        </tr>
                    `;
                });
                updateStats();
            });
        }

        // ⏳ রিকোয়েস্ট লোড
        function loadRequests() {
            db.collection('requests').orderBy('createdAt', 'desc').onSnapshot((snapshot) => {
                const tbody = document.getElementById('requestsTableBody');
                tbody.innerHTML = '';
                let i = 1;
                snapshot.forEach((doc) => {
                    const data = doc.data();
                    const statusClass = data.status === 'Pending' ? 'badge-pending' : 'badge-active';
                    tbody.innerHTML += `
                        <tr>
                            <td>${i++}</td>
                            <td>${data.student}</td>
                            <td>${data.subject}</td>
                            <td>${data.location}</td>
                            <td><span class="badge-status ${statusClass}">${data.status}</span></td>
                            <td>

                                ` : '✅ Done'}
                            </td>
                        </tr>
                    `;
                });
                updateStats();
            });
        }

        // 📊 পরিসংখ্যান আপডেট
        function updateStats() {
            db.collection('tutors').get().then((snap) => {
                document.getElementById('totalTutors').textContent = snap.size;
                let active = 0;
                snap.forEach(doc => { if(doc.data().status === 'Active') active++; });
                document.getElementById('activeTutors').textContent = active;
            });
            db.collection('students').get().then((snap) => {
                document.getElementById('totalStudents').textContent = snap.size;
            });
            db.collection('requests').where('status', '==', 'Pending').get().then((snap) => {
                document.getElementById('pendingRequests').textContent = snap.size;
            });
        }

        function saveTutor() {
            const name = document.getElementById('tutorName').value.trim();
            const subject = document.getElementById('tutorSubject').value.trim();
            const status = document.getElementById('tutorStatus').value;
            if(!name || !subject) return alert('সব ফিল্ড পূরণ করুন!');

            db.collection('tutors').add({
                name, subject, status,
                createdAt: firebase.firestore.FieldValue.serverTimestamp()
            }).then(() => {
                closeModal('addTutorModal');
                document.getElementById('tutorName').value = '';
                document.getElementById('tutorSubject').value = '';
                loadTutors();
            });
        }

        // 🟢 স্টুডেন্ট সেভ
        function saveStudent() {
            const name = document.getElementById('studentName').value.trim();
            const cls = document.getElementById('studentClass').value.trim();
            const tutor = document.getElementById('studentTutor').value.trim();
            const status = document.getElementById('studentStatus').value;
            if(!name || !cls) return alert('সব ফিল্ড পূরণ করুন!');

            db.collection('students').add({
                name, class: cls, tutor, status,
                createdAt: firebase.firestore.FieldValue.serverTimestamp()
            }).then(() => {
                closeModal('addStudentModal');
                document.getElementById('studentName').value = '';
                document.getElementById('studentClass').value = '';
                document.getElementById('studentTutor').value = '';
                loadStudents();
            });
        }

        // 🗑️ ডিলিট টিউটর
        function deleteTutor(id) {
            if(confirm('টিউটরটি ডিলিট করতে চান?')) {
                db.collection('tutors').doc(id).delete().then(() => loadTutors());
            }
        }

            }
        }

        // 🔑 পাসওয়ার্ড পরিবর্তন (ডেমো)
        function changeAdminPassword() {
            const newPass = document.getElementById('newAdminPass').value.trim();
            if(!newPass || newPass.length < 6) return alert('পাসওয়ার্ড কমপক্ষে ৬ অক্ষরের হতে হবে!');
            const user = auth.currentUser;
            if(user) {
                user.updatePassword(newPass).then(() => {
                    alert('পাসওয়ার্ড সফলভাবে পরিবর্তন হয়েছে!');
                    document.getElementById('newAdminPass').value = '';
                }).catch(() => alert('পাসওয়ার্ড পরিবর্তনে সমস্যা!'));
            } else {
                alert('আপনি লগইন নেই!');
            }
        }
